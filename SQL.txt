CREATE DATABASE HOJA_VIDA;
use HOJA_VIDA;

CREATE TABLE DATOS_PERSONALES
(
PK_DAT_PERSONALES INT NOT NULL AUTO_INCREMENT,
NOMBRES VARCHAR(50),
APELLIDOS VARCHAR(50),
LUGAR_NAC VARCHAR(20),
FECHA_NAC VARCHAR(10),
EDAD INT,
DISCAPACIDAD VARCHAR(2),
CERT_DISCAPACIDAD VARCHAR(300),
PRIMARY KEY(PK_DAT_PERSONALES)
);

CREATE TABLE DATOS_POSTULANTE
(
PK_DAT_POSTULANTE INT NOT NULL AUTO_INCREMENT,
CARGO_A_POSTULAR VARCHAR(100),
CONDICION VARCHAR(80),
TIPO_CURSO VARCHAR(8),
NOTA DOUBLE,
PRE_PROMEDIO DOUBLE,
ORDEN_MERITO INT,
PLAZAS_VACANTES INT,
TIPO_CALIFICACION VARCHAR(30),
PK_DAT_PERSONALES INT,
PRIMARY KEY(PK_DAT_POSTULANTE),
INDEX PERSONALESPOSTULANTE(PK_DAT_PERSONALES),
FOREIGN KEY(PK_DAT_PERSONALES) REFERENCES DATOS_PERSONALES(PK_DAT_PERSONALES)
);

CREATE TABLE CONVOCATORIAS_ANTERIORES(
CONVOCATORIA VARCHAR(50),
PLAZA VARCHAR(50),
ETAPA VARCHAR(50),
PK_DAT_POSTULANTE INT,
INDEX POSTULANTECONVOCATORIAS(PK_DAT_POSTULANTE),
FOREIGN KEY(PK_DAT_POSTULANTE) REFERENCES DATOS_POSTULANTE(PK_DAT_POSTULANTE)
);



CREATE TABLE DATOS_ACADEMICOS(
PK_DAT_ACADEMICOS INT NOT NULL AUTO_INCREMENT,
UNIV_PROCEDENCIA VARCHAR(80),
COLEGIO_PROFESIONAL VARCHAR(80),
FECHA_INCORPORACION VARCHAR(10),
TITULO_OTROS VARCHAR(100),
TESIS_TITULAR VARCHAR(100),
PK_DAT_PERSONALES INT,
PRIMARY KEY(PK_DAT_ACADEMICOS),
INDEX PERSONALESACADEMICOS(PK_DAT_PERSONALES),
FOREIGN KEY(PK_DAT_PERSONALES) REFERENCES DATOS_PERSONALES(PK_DAT_PERSONALES)
);


CREATE TABLE ESTUDIOS_PROFESIONALES(
DET_ESTUDIOS VARCHAR(50),
UNIVERSIDAD VARCHAR(80),
NIVEL VARCHAR(30),
MENCION VARCHAR(50),
ANIO VARCHAR(4),
PK_DAT_ACADEMICOS INT,
INDEX ACADEMICOSPROFESIONALES(PK_DAT_ACADEMICOS),
FOREIGN KEY(PK_DAT_ACADEMICOS) REFERENCES DATOS_ACADEMICOS(PK_DAT_ACADEMICOS)
);

CREATE TABLE MERITOS_UNIV(
PUESTO VARCHAR(20),
UNIVERSIDAD VARCHAR(80),
ANIO VARCHAR(4),
PK_DAT_ACADEMICOS INT,
INDEX ACADEMICOSMERITOS(PK_DAT_ACADEMICOS),
FOREIGN KEY(PK_DAT_ACADEMICOS) REFERENCES DATOS_ACADEMICOS(PK_DAT_ACADEMICOS)
);

CREATE TABLE EXPERIENCIA_PROFESIONAL(
PK_EXP_PROFESIONAL INT NOT NULL AUTO_INCREMENT,
PK_DAT_PERSONALES INT,
PRIMARY KEY(PK_EXP_PROFESIONAL),
INDEX PERSONALESEXPERIENCIA(PK_DAT_PERSONALES),
FOREIGN KEY(PK_DAT_PERSONALES) REFERENCES DATOS_PERSONALES(PK_DAT_PERSONALES)
);

CREATE TABLE DOC_DES_PROFESIONAL(
NUM_EXP INT,
DDAS VARCHAR(30),
DDAE VARCHAR(30),
MATERIA VARCHAR (30),
ESPECIALIDAD VARCHAR(30),
CALIFICACION1 DOUBLE,

PK_EXP_PROFESIONAL INT, 
INDEX EXPERIENCIADOCDES(PK_EXP_PROFESIONAL),
FOREIGN KEY(PK_EXP_PROFESIONAL) REFERENCES EXPERIENCIA_PROFESIONAL(PK_EXP_PROFESIONAL)
);

CREATE TABLE CAL_JUD_FIS(

DDAGRAVIADO VARCHAR(30),

MATERIA VARCHAR(30),

PK_EXP_PROFESIONAL INT,
INDEX EXPERIENCIACALJUD(PK_EXP_PROFESIONAL),
FOREIGN KEY(PK_EXP_PROFESIONAL) REFERENCES EXPERIENCIA_PROFESIONAL(PK_EXP_PROFESIONAL)
);

CREATE TABLE MAGISTRADO_RATIFICADO(
RESPUESTA1 VARCHAR(2),
RESPUESTA2 VARCHAR(30),

);

CREATE TABLE SITUACION_LABORAL(
PK_SIT_LABORAL INT NOT NULL AUTO_INCREMENT,

INDEX EXPERIENCIASITLAB(PK_EXP_PROFESIONAL),
